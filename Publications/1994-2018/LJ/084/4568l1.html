#!/usr/bin/perl
#
# Local configuration parameters for intranet
# installation script
#
# Set the path to whatever your apache server root is.
# On Red Hat, this may be /home/httpd
# The standard source build from Apache uses
# /usr/local/apache
#
# You will also need to know the document root:
# this is usually document_root/htdocs or
# document_root/html
#
#
$apache_path="/usr/local/apache";
$apache_doc_root="/usr/local/apache/htdocs";
$apache_cgi_bin="/usr/local/apache/cgi-bin";
#
