#!/usr/local/bin/python
#

# Echo client program

from socket import *
HOST  = 'daring.cwi.nl'    # The remote host
PORT  =  50007          # Arbitrary non-privileged server
s  = socket( AF_INET, SOCK_STREAM )
s.connect( HOST, PORT )
s.send( 'Hello, world' )
data =  s.recv( 1024 )
s.close()
print 'Received', data
