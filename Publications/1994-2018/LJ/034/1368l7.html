#!/usr/local/bin/python
#
#

import cgi


def print_header() :
    ############
   print "Content-type: text/html"
   print
   print "<HTML>"
   print "<Head><Title>"
   print "Parameters Echo"
   print "</Title></Head>"
   print "<Body><H1>"
   print "Parameters Echo</H1><hr>"


def display_query( form ) :
    #############
   fields = form.keys()
   print "<P><H2>The following"
   print "parameters were received:</H2>"
   print "<UL>"
   if len( fields ) = 0 :
        print "<LI>Nothing"
   else :
        for f in fields :
             print "<LI>", f, " : ", cgi.escape( form[ f ] )
   print "</UL>"


def print_footer() :
    ############
   print "</Body></HTML>"


def Main() :
    ####
   form = cgi.SvFormContentDict()
   print_header()
   display_query( form )
   print_footer()


Main()
