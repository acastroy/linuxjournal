Linux:

gcc -DLINUX -o integrator integrator.c -lm
Cygwin: 
gcc -DCYGWIN -o integrator.exe integrator.c \
   -lm -lwinmm
IRIX: 
gcc -DIRIX -o integrator integrator.c -lm -laudio

wish -f duffing2.tcl | integrator |
