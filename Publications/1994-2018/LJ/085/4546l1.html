### lilo.conf on PC1 ###
boot=/dev/hda
map=/boot/map
install=/boot/boot.b
prompt
timeout=50
linear
default=mosix
# Red Hat Default Kernel
image=/boot/vmlinuz-2.2.14-5.0
        label=original
        read-only
        root=/dev/hda1
# MOSIX Kernel, booted by default
image=/boot/bzImage.mosix
        label=mosix
        read-only
        root=/dev/hda1
