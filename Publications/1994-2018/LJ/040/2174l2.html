#!/bin/sh
#
# XATTRD   Example file to build rc scripts.
#
# Source function library.
# This is for Red Hat system only
 . /etc/rc.d/init.d/functions
#
# See how we were called.
case "$1" in
 start)
    if [ -f /usr/sbin/xattrd ]; then
        touch /var/lock/subsys/xattrd
        daemon /usr/sbin/xattrd
    fi
    ;;
 stop)
    killall xattrd
    rm -f /var/lock/subsys/xattrd
    ;;
 *)
    echo "Usage: xattrd {start|stop}"
    exit 1
esac

exit 0
