Listing 3. elisp Code

;;[this is the elisp code that defines these shell
;;scripts as externals
;;for emacs' tex-mode commands.]

(defun tex-local-setup ()
  "This is the local setup for plain-TeX and LaTeX
   modes. Note that this function is given as
   the FIRST defininition of tex-mode-hook."
    (setq tex-dvi-view-command "gspreview")
    (setq tex-dvi-print-command "gsprint"))

(setq-default tex-mode-hook 'tex-local-setup)
