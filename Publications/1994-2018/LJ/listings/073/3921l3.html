Listing 3. New On-Line Rendering

RtpMapping RtpRender::newOnlineRender(BBox& 
   viewPort, QPixmap *buf)
{
  if (_privateBuff) {      // Cancel any private 
                           // rendering
    delete _privateBuff;
    _privateBuff = NULL;
  }
  _buf = buf;
  startRender(viewPort);   // startRender fills 
                           // in _map
  signalRepaintWindow();
  return _map;
}
