Listing 3. Proxy ARP

# Bridging only works with Ethernet, so I have
# to do something else to get the same effect with <\n>
# PLIP. Fortunately, there's proxyarp! I'll just<\n>
# tell eth1 to tell the world that it is
# acting on behalf of the
# machine on the other end of the PLIP connection,<\n>
# and it should work finei.
[ $KVERSION = "2.2" ] && {
# 2.2 requires me to explicitly enable this feature
echo 1 > /proc/sys/net/ipv4/conf/eth1/proxy_arp
}
arp -i eth1 -s egg eth1 pub
