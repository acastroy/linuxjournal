Listing 1. Creation of Local Variables

addpath()
{
# local variables
typeset options_shift_val options_missing_arg\
   options_unknown_opt
typeset options_num_args_left
typeset opt_h opt_f opt_b opt_p
typeset pathvar sep COMMAND dirname tmp element
typeset USAGE="addpath [-h] [-f|-b] [-p\
   <pathvar>] [dirname]"
typeset USAGE1="-f adds [dirname] to front of \
   <pathvar> (default: PATH)"
typeset USAGE2="-b adds [dirname] to back of \
   <pathvar> (default: PATH)"
