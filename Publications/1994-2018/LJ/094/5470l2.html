Listing 2. /etc/vpnd/vpnd.conf

# change the mode to "client" on the client. Otherwise
# this file is the same on client and server.
mode server

# "real" IP addresses for the WiFi network.
client 10.2.0.2 2001
server 10.2.0.1 2001

# virtual IP addresses
remote 10.3.0.2
local 10.3.0.1
# location of the VPN key, and where vpnd should write its process
# ID
keyfile /etc/vpnd/vpnd.key
pidfile /var/run/vpnd.pid

keyttl 120
randomdev /dev/urandom
keepalive 10
noanswer 3
mtu 1500
