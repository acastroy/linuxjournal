Listing 1. /etc/pcmcia/network.opts

# network.opts file for base station
# Put in your own MAC and IP addresses.

case "$ADDRESS" in

*,*,*,00:02:2D:27:29:8D)
# WiFi card
    IPADDR="10.2.0.1"
    NETMASK="255.255.0.0"
    ;;

*,*,*,00:50:04:94:E5:6B)
# 10base-T Ethernet card
    IPADDR="198.144.202.252"
    NETMASK="255.255.255.248"
    GATEWAY="198.144.202.249"
    ;;

esac
