Listing 1. Nested Loops in the mcERP Algorithm

for i = 1:M
   for j = N:-1:1
      norm = 0;
      for r = 1:R
         for t = 1:T
            X = x(i,r,t);
            for n = 1:N
   if (n ~= j)
       X = X - coupling(i,n) *
                    alpha(n,r) *
                    s (n,pad+t-tau(n,r));
      end
            end
         Y = alpha(j,r)*s(j,pad+t-tau(j,r));
         c(i,j) = c(i,j) + X*Y;
         norm = norm + Y^2;
       end
    end
    c(i,j) = c(i,j)/norm;
  end
end
