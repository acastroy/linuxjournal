Listing 1. /etc/ipsec.conf on a Wireless Client

(Note: both leftrsasigkey and rightrsasigkey have been truncated.)

# basic configuration
config setup
    interfaces=%defaultroute
    klipsdebug=none
    plutodebug=none
    plutoload=%search
    plutostart=%search
    uniqueids=yes

conn %default
    keyingtries=0
    authby=secret

conn george-gracie
    authby=rsasig
    left=10.0.54.2
    leftid=@george.wiremonkeys.org
    leftsubnet=0.0.0.0/0
    leftrsasigkey=0sAQPF0JJvY7...
    right=%defaultroute
    rightid=@gracie.wiremonkeys.org
    rightrsasigkey=0sNU0q2Y0Y0jxAIKuutV3...
    auto=start
