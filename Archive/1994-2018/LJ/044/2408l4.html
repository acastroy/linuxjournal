Listing 4. TIS FWTK config file, /usr/local/etc/netperm-table

#
# Config file for TIS fwtk http-gw HTTP proxy
# server and plug-gw for NNTP
# on a class C internal network with bogus IP
# address 192.168.50
#
http-gw: userid uucp
http-gw: groupid 6
http-gw: default-httpd www.tis.com:80
http-gw: permit-hosts 127.0.0.1 -log all
http-gw: permit-hosts 192.168.50.* -log all
http-gw: deny-hosts unknown
http-gw: hosts 192.168.50.*
plug-gw: timeout 1800
plug-gw: port nntp 192.168.50.* -plug-to \
   news_IP_address -port 119
