Listing 1. Excerpt of the Parameter file to Configure the Device Drivers

#ifndef XPARAMETERS_H    /* prevent circular
                            inclusions */
#define XPARAMETERS_H    /* by using protection
                            macros */

/******************************************
 *
 * Interrupt Controller (Intc) defines.
 * DeviceID starts at 0
 *

#define XPAR_INTC_NUM_INSTANCES  1  /* Number of
                                       instances*/

#define XPAR_INTC_0_DEVICE_ID    1  /* Device ID for
                                       instance */
#define XPAR_INTC_0_MAX_ID      31  /* only 32
                                       interrupts
                                       currently */
                                    /* valid range
                                       is 0-31 */
#define XPAR_INTC_0_REG_BASE    0xD0000FC0
                                /* Register base
                                   address */
#define XPAR_INTC_0_UARTNS550_0_VEC_ID 31
#define XPAR_INTC_0_IIC_0_VEC_ID       29
#define XPAR_INTC_0_EMAC_0_VEC_ID      28

#define XPAR_INTC_MAX_ID        31  /* valid range
                                       is 0-31 */

#endif              /* end of protection macro */
