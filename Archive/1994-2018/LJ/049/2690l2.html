Listing 2. Calculator.fv

(Shape (Width 205 + Inf) (Height 220 + Inf)
  (Color       "black")
  (BgColor     "rather dark blue")
  (LightShadow "somewhat light steel blue")
  (DarkShadow  "very dark steel blue")

  (Macro BTN (label)
     (Border (Pen 1)
       (Shape
       (Width  50)
       (Height 40)
       (BgColor "white")
       (Frame Raised (Button (ShadowSize 0) label))
     )))

  (VBox
    (Shape (BgColor "light steel blue") (Text
   "Simple Calculator GUI Form"))
    (Fill)
    (Border (Pen 4) (Shape (BgColor
   "light yellow") (Text
RightAlign "0. ")))
    (Fill)
    (HBox (ShadowSize 8) (BTN "7") (BTN "8")
   (BTN "9") (Fill) (BTN "X"))
    (HBox (ShadowSize 8) (BTN "4") (BTN "5")
   (BTN "6") (Fill) (BTN "/"))
    (HBox (ShadowSize 8) (BTN "1") (BTN "2")
   (BTN "3") (Fill) (BTN "+"))
    (HBox (ShadowSize 8) (BTN "0") (BTN ".")
   (BTN "=") (Fill) (BTN "-"))
  ))
